## cntX + cntO (optimal)
```java
class Solution {
    public boolean validTicTacToe(String[] board) {
        int cntX = 0, cntO = 0;
        for (String row : board) {
            for (char c : row.toCharArray()) {
                if (c == 'X') cntX++;
                if (c == 'O') cntO++;
            }
        }
        if (cntO != cntX && cntO != cntX - 1) return false;
        if (win(board) == 'X') return cntO == cntX - 1;
        if (win(board) == 'O') return cntO == cntX;
        return true;
    }
    private char win(String[] board) {
        int[] rows = new int[3];
        int[] cols = new int[3];
        int diagonal = 0;
        int antiDiagonal = 0;
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                if (board[i].charAt(j) == 'X') {
                    rows[i]++;
                    cols[j]++;
                    if (i == j) diagonal++;
                    if (i + j == 2) antiDiagonal++;
                } else if (board[i].charAt(j) == 'O') {
                    rows[i]--;
                    cols[j]--;
                    if (i == j) diagonal--;
                    if (i + j == 2) antiDiagonal--;
                }
                if (Math.abs(rows[i]) == 3 || Math.abs(cols[j]) == 3 ||
                    Math.abs(diagonal) == 3 || Math.abs(antiDiagonal) == 3) {
                    return board[i].charAt(j);
                }
            }
        }
        return ' ';
    }
}
```
