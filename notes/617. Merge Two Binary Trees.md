[Leetcode](https://leetcode.com/problems/merge-two-binary-trees/)

## dfs(bottom up) (optimal)
```java
class Solution {
    public TreeNode mergeTrees(TreeNode t1, TreeNode t2) {
        if (t1 == null) return t2;
        if (t2 == null) return t1;
        t1.left = mergeTrees(t1.left, t2.left);
        t1.right = mergeTrees(t1.right, t2.right);
        t1.val += t2.val;
        return t1;
    }
}
```

## dfs(top down) (optimal)
```java
class Solution {
    public TreeNode mergeTrees(TreeNode t1, TreeNode t2) {
        if (t1 == null) return t2;
        if (t2 == null) return t1;
        t1.val += t2.val;
        t1.left = mergeTrees(t1.left, t2.left);
        t1.right = mergeTrees(t1.right, t2.right);
        return t1;
    }
}
```

## stack (optimal)
```java
class Solution {
    public TreeNode mergeTrees(TreeNode t1, TreeNode t2) {
        if (t1 == null) return t2;
        Stack<TreeNode[]> stack = new Stack<>();
        stack.add(new TreeNode[]{t1, t2});
        while (!stack.isEmpty()) {
            TreeNode[] cur = stack.pop();
            if (cur[1] == null) continue;
            cur[0].val += cur[1].val;
            if (cur[0].left == null) cur[0].left = cur[1].left;
            else stack.add(new TreeNode[]{cur[0].left, cur[1].left});
            if (cur[0].right == null) cur[0].right = cur[1].right;
            else stack.add(new TreeNode[]{cur[0].right, cur[1].right});
        }
        return t1;
    }
}
```
