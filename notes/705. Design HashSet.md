## int array (basic)
```java
class MyHashSet {
    private int[] set;
    public MyHashSet() {
        set = new int[10000000];
    }
    
    public void add(int key) {
        set[key] = 1;
    }
    
    public void remove(int key) {
        set[key] = 0;
    }
    
    public boolean contains(int key) {
        return set[key] == 1;
    }
}
```

## `List<Integer>[]` container + rehash (optimal)
```java
class MyHashSet {
    private int cap;
    private List<Integer>[] container;
    private double loadFactor;
    private int cnt;
    public MyHashSet() {
        cap = 1000;
        container = new LinkedList[cap];
        loadFactor = 0.75;
        cnt = 0;
    }
    public void add(int key) {
        if (contains(key)) return;
        // rehash
        if (cap * loadFactor == cnt) {
            cnt = 0;
            cap *= 2;
            List<Integer>[] oldContainer = container;
            container = new LinkedList[cap];
            for (int i = 0; i < oldContainer.length; i++) {
                List<Integer> list = oldContainer[i];
                if (list != null) {
                    for (int val : list) {
                        this.add(val);
                    }
                }
            }
        }
        int code = key % cap;
        if (container[code] == null) {
            container[code] = new LinkedList<>();
        }
        container[code].add(key);
        cnt++;
    }
    public void remove(int key) {
        int code = key % cap;
        List<Integer> list = container[code];
        if (list != null) {
            Iterator<Integer> itr = list.iterator();
            while (itr.hasNext()) {
                if (itr.next() == key) {
                    itr.remove();
                    cnt--;
                    break;
                }
            }
        }
    }
    public boolean contains(int key) {
        int code = key % cap;
        List<Integer> list = container[code];
        if (list != null) {
            Iterator<Integer> itr = list.iterator();
            while (itr.hasNext()) {
                if (itr.next() == key) {
                    return true;
                }
            }
        }
        return false;
    }
}
```
