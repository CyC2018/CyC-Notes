## hashmap (basic)
```java
class FileSystem {
    private Map<String, Integer> map;
    public FileSystem() {
        map = new HashMap<>();
        map.put("", -1);
    }
    public boolean createPath(String path, int value) {
        // find parent position
        int index = path.lastIndexOf("/");
        String pred = path.substring(0, index);
        if (!map.containsKey(pred)) return false;
        // check if already exists
        if (map.containsKey(path)) return false;
        // insert
        map.putIfAbsent(path, value);
        // return true
        return true;
    }
    public int get(String path) {
        return map.getOrDefault(path, -1);
    }
}
```

## trie tree node (optimal)
```java
class FileSystem {
    private class File {
        String name;
        int val = -1;
        Map<String, File> map = new HashMap<>();
        File(String name) {
            this.name = name;
        }
        File(String name, int val) {
            this.name = name;
            this.val = val;
        }
    }
    File root;
    public FileSystem() {
        root = new File("");
    }
    
    public boolean createPath(String path, int value) {
        String[] arr = path.split("/");
        System.out.println(Arrays.toString(arr));
        // find parent position
        File pred = root;
        for (int i = 1; i < arr.length - 1; i++) {
            String nextName = arr[i];
            if (!pred.map.containsKey(nextName)) {
                return false;
            }
            pred = pred.map.get(nextName);
        }
        // check if already exists
        String name = arr[arr.length - 1];
        if (pred.map.containsKey(name)) return false;
        // create
        File toAdd = new File(name, value);
        // insert
        pred.map.put(name, toAdd);
        // return true
        return true;
    }
    
    public int get(String path) {
        String[] arr = path.split("/");
        File cur = root;
        for (int i = 1; i < arr.length; i++) {
            String nextName = arr[i];
            if (!cur.map.containsKey(nextName)) {
                return -1;
            }
            cur = cur.map.get(nextName);
        }
        return cur.val;
    }
}
```
