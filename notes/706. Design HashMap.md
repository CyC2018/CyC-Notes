## int array (basic)
```java
class MyHashMap {
    private int[] map;
    public MyHashMap() {
        map = new int[100000];
        Arrays.fill(map, -1);
    }
    public void put(int key, int value) {
        map[key] = value;
    }
    public int get(int key) {
        return map[key];
    }
    public void remove(int key) {
        map[key] = -1;
    }
}
```
## `List<Integer>[]` container + rehash (optimal)
```java
class MyHashMap {
    private int cap;
    private List<int[]>[] container;
    private double loadFactor;
    private int cnt;
    public MyHashMap() {
        cap = 1000;
        container = new LinkedList[cap];
        loadFactor = 0.75;
        cnt = 0;
    }
    public void put(int key, int value) {
        // update
        if (get(key) != -1) {
            int code = key % cap;
            if (container[code] != null) {
                for (int[] cur : container[code]) {
                    if (cur[0] == key) {
                        cur[1] = value;
                        return;
                    }
                }
            }
        }
        // rehash
        if (cnt == cap * loadFactor) {
            cnt = 0;
            cap *= 2;
            List<int[]>[] oldC = container;
            container = new List[cap];
            for (List<int[]> list : oldC) {
                if (list != null) {
                    for (int[] cur : list) {
                        this.put(cur[0], cur[1]);
                    }
                }
            }
        }
        // insert
        int code = key % cap;
        if (container[code] == null) {
            container[code] = new LinkedList<>();
        }
        container[code].add(new int[]{key, value});
        cnt++;
    }
    public int get(int key) {
        int code = key % cap;
        List<int[]> list = container[code];
        if (list != null) {
            for (int[] cur : list) {
                if (cur[0] == key) {
                    return cur[1];
                }
            }
        }
        return -1;
    }
    public void remove(int key) {
        int code = key % cap;
        List<int[]> list = container[code];
        if (list != null) {
            for (int i = 0; i < list.size(); i++) {
                if (list.get(i)[0] == key) {
                    list.remove(i);
                    return;
                }
            }
        }
    }
}
```
