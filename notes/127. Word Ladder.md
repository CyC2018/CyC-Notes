[Leetcode](https://leetcode.com/problems/word-ladder/)

## bfs

时间O(N^2 * M)，空间O(N)。N是wordList长度，M是单词长度。

```java
class Solution {
    public int ladderLength(String beginWord, String endWord, List<String> wordList) {
        // 把起始单词也加入单词列表，为了后面一起建立邻接图
        wordList.add(beginWord);
        // 记录起始单词的位置，为了后面放进队列
        int start = wordList.size() - 1;
        // 检查列表中是否有结束单词
        int end = -1;
        for (int i = 0; i < wordList.size(); i++) {
            if (wordList.get(i).equals(endWord)) end = i;
        }
        if (end == -1) return 0;
        // 根据单词列表建立邻接图（图中只保存单词在列表里位置）
        List<Integer>[] graph = genGraph(wordList);
        // 定义标记数组
        boolean[] marked = new boolean[wordList.size()];
        // 定义队列
        Queue<Integer> queue = new LinkedList<>();
        // 起始单词位置加入队列
        queue.add(start);
        // 起始单词标记已访问，避免回访无限循环
        marked[start] = true;
        // 定义路径长度
        int len = 1;
        // 开始循环
        while (!queue.isEmpty()) {
            // 记录当前大小
            int size = queue.size();
            // 遍历并清空当前队列
            for (int i = 0; i < size; i++) {
                // 取出一个
                int cur = queue.poll();
                // 判断是否已经到达终点
                if (cur == end) return len;
                // 根据已经建好的图遍历所有可能合法方向
                for (int next : graph[cur]) {
                    if (marked[next] == false) {
                        queue.add(next);
                        marked[next] = true;
                    }
                }
            }
            // 路径长度增加1
            len++;
        }
        // 找不到路径返回0
        return 0;
    }
    
    private List<Integer>[] genGraph(List<String> wordList) {
        List<Integer>[] graph = new List[wordList.size()];
        for (int i = 0; i < wordList.size(); i++) {
            graph[i] = new ArrayList<>();
            for (int j = 0; j < wordList.size(); j++) {
                if (i != j && isConnected(wordList.get(i), wordList.get(j))) {
                    graph[i].add(j);
                }
            }
        }
        return graph;
    }
                    
    private boolean isConnected(String s1, String s2) {
        int cnt = 0;
        for (int i = 0; i < s1.length(); i++) {
            if (s1.charAt(i) != s2.charAt(i)) cnt++;
        }
        return cnt == 1;
    }
}
```

不建立邻接图，直接用26个字符试。时间O(N * M^2)，空间O(NM)。N是wordList长度，M是单词长度。

```java
class Solution {
    public int ladderLength(String beginWord, String endWord, List<String> wordList) {
        // 全都放入set里，实现O(1)查询
        HashSet<String> set = new HashSet<>(wordList);
        // 目标单词不存在，无解
        if (!set.contains(endWord)) return 0;
        // 定义队列
        Queue<String> queue = new LinkedList<>();
        // 起始单词加入队列
        queue.add(beginWord);
        // 定义路径长度
        int len = 1;
        // 开始循环
        while (!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                String curStr = queue.poll();
                if (curStr.equals(endWord)) return len;
                char[] cur = curStr.toCharArray();
                // 从队列中拿出来的单词，只允许改一个字符，然后看是否在set中
                for (int j = 0; j < cur.length; j++) {
                    char[] next = cur.clone();
                    for (char c = 'a'; c <= 'z'; c++) {
                        next[j] = c;
                        String nextStr = new String(next);
                        // 如果在set中，就加入队列，并从set中去除防止再次遍历
                        if (set.contains(nextStr)) {
                            queue.add(nextStr);
                            set.remove(nextStr);
                        }
                    }
                }
            }
            len++;
        }
        return 0;
    }
}
```
